234// hist
235std::tuple<std::vector<double>, std::vector<double>, container::BarContainer>
236Axes::hist(const pybind11::tuple &args, const pybind11::dict &kwargs) {
237 pybind11::list ret = hist_attr(*args, **kwargs);
238 // parse N, bins, patches
239 pybind11::list N_obj = ret[0], bins_obj = ret[1];
240 pybind11::object patches_obj = ret[2];
241 std::vector<double> N, bins;
242 for (auto it = N_obj.begin(); it != N_obj.end(); it++) {
243 N.push_back(it->cast<double>());
244 }
245 for (auto it = bins_obj.begin(); it != bins_obj.end(); it++) {
246 bins.push_back(it->cast<double>());
247 }
248 return {N, bins, container::BarContainer(patches_obj)};
249}
