162std::tuple<figure::Figure, std::vector<axes::Axes>>
163PyPlot::subplots(int r, int c,
164 const pybind11::dict &kwargs = pybind11::dict()) {
165 pybind11::tuple args = pybind11::make_tuple(r, c);
166 pybind11::list ret = subplots_attr(*args, **kwargs);
167 std::vector<axes::Axes> axes;
168 pybind11::object fig = ret[0];
169 figure::Figure figure(fig);
170 if (r == 1 and c == 1) {
171 pybind11::object ax = ret[1];
172 axes.push_back(axes::Axes(ax));
173 } else {
174 pybind11::list axs = ret[1];
175 for (pybind11::size_t i = 0; i < axs.size(); ++i) {
176 axes.push_back(axes::Axes(axs[i]));
177 }
178 }
179 return {figure, axes};
180}
